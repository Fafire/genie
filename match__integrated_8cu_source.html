100 int getBitmapSize(int &in_out_bitmap_bits, u32 in_shift_bits_subsequence, int in_number_of_data_points, int in_queries_size)
101 {
102  if (in_out_bitmap_bits < 0)
103  in_out_bitmap_bits = -in_out_bitmap_bits; //for hash_table_size, still let it determine by users currently
104 
105  int threshold = in_out_bitmap_bits - 1, bitmap_size = 0;
106  if (in_out_bitmap_bits > 1)
107  {
108  float logresult = std::log2((float) in_out_bitmap_bits);
109  in_out_bitmap_bits = (int) logresult;
110  if (logresult - in_out_bitmap_bits > 0)
111  {
112  in_out_bitmap_bits += 1;
113  }
114  logresult = std::log2((float) in_out_bitmap_bits);
115  in_out_bitmap_bits = (int) logresult;
116  if (logresult - in_out_bitmap_bits > 0)
117  {
118  in_out_bitmap_bits += 1;
119  }
120  in_out_bitmap_bits = pow(2, in_out_bitmap_bits);
121  bitmap_size = ((((unsigned int)1<<in_shift_bits_subsequence) * in_number_of_data_points) / (32 / in_out_bitmap_bits) + 1)
122  * in_queries_size;
123  }
124  else
125  {
126  in_out_bitmap_bits = threshold = 0;
127  }
128 
129  Logger::log(Logger::DEBUG, "Bitmap bits: %d, threshold:%d, shift_bits_subsequence: %d",
130  in_out_bitmap_bits, threshold, in_shift_bits_subsequence);
131 
132  return bitmap_size;
133 }
